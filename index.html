Service Workers!

addEventListener('fetch', e => {
  // e.respondWith accepts a Promise which should resolve to a Response
  e.respondWith(
    (async () => {
      // Check to see if we have a cached version of the response matching
      // this request
      const cache = await Cache.open('my-cache')
      const response = cache.match(e.request)
      if (response) return response

      // If not, let's attempt the original request
      return fetch(e.request)
    })(),
  )
})
const staticFiles = ['/vendor.js', '/bundle.js', '/client.css']

addEventListener('install', (e: InstallEvent) => {
  // Cache all static assets
  e.waitUntil(
    (async () => {
      const cache = await getCache()
      return cache.addAll(staticFiles)
    })(),
  )
})
addEventListener('fetch', (e: FetchEvent) => {
  e.respondWith(handleRequest(e.request))
})

async function handleRequest(req: Request) {
  const cache = await getCache()

  // For any /budgets/:id/* URL, load the budgets application shell
  if (req.url.indexOf('/budgets/') > -1) {
    const url = new URL(req.url)
    const id = url.pathname.split('/')[2]

    if (id) {
      url.pathname = '/budgets/' + id
      const res = await cache.match(
        new Request(url.toString(), { headers: req.headers }),
      )
      if (res) return res
    }
  }

  const res = await caches.match(req)
  if (res) return res

  return fetch(req)
}
